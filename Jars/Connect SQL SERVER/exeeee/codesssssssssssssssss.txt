    void selection_modifier() {
        try {

            int row = tClient.getSelectedRow();
            String Table_click = (tClient.getModel().getValueAt(row, 0).toString());
            Connection con = DriverManager.getConnection(ClsConnexion.getConnection());
            String sql = "select * from vTableClient where id = '" + Table_click + "'";
            PreparedStatement pst = con.prepareStatement(sql);
            ResultSet rs = pst.executeQuery();
            if (rs.next()) {
                String id = rs.getString("id");
                String nom=rs.getString("nom");
                String prenom = rs.getString("prenom");
                String numcarte =rs.getString("numerocarteidentite");
             
                
                
                 
               tId.setText(id);
               jNom.setText(nom);
               jPrenom.setText(prenom);
               jCarte.setText(numcarte);
               
            } 
            con.close();

        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);
        }
    }
    
    void ChargerComboBoxQuartier() {
        try {
            
            Connection con = DriverManager.getConnection(ClsConnexion.getConnection());
            String sql = "select designation from tQuartier";
            PreparedStatement pst = con.prepareStatement(sql);
            ResultSet rs = pst.executeQuery();
            while (rs.next()) {
                jQuartier.addItem(rs.getString("designation"));                 
            } 
            con.close();

        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);
        }
    }
